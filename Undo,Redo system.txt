class TextEditor:
    def __init__(self):
        self.document = ""
        self.undo_stack = []
        self.redo_stack = []

    def make_change(self, new_text):
        self.undo_stack.append(self.document)
        self.redo_stack.clear()
        self.document = new_text
        print("Change made.")

    def undo(self):
        if not self.undo_stack:
            print("Nothing to undo.")
            return
        self.redo_stack.append(self.document)
        self.document = self.undo_stack.pop()
        print("Undo performed.")

    def redo(self):
        if not self.redo_stack:
            print("Nothing to redo.")
            return
        self.undo_stack.append(self.document)
        self.document = self.redo_stack.pop()
        print("Redo performed.")

    def display(self):
        print("Current Document State:", self.document)


editor = TextEditor()
editor.make_change("Hello")
editor.display()
editor.make_change("Hello, World!")
editor.display()
editor.make_change("Hello, World! This is GPT.")
editor.display()
editor.undo()
editor.display()
editor.undo()
editor.display()
editor.redo()
editor.display()
